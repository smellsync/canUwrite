    <p>Whatever you type will appear as the letter two places after in the alphabet.</p>

    <p>To form the right message, type two letters before your target letter.</p>

    <div id="output"></div>

    <script>

        document.addEventListener("keydown", function(event) {

            let char = event.key;

            if (char.length === 1 && char.match(/[a-zA-Z]/)) {

                let shiftedChar = String.fromCharCode(char.charCodeAt(0) + 2);

                document.getElementById("output").innerText += shiftedChar;

            }

        });

    </script>
